% Optimization

\makeatletter
\newcommand{\optimization}{\@ifstar{\@nonumberoptimization}{\@numberedoptimization}}
\newcommand{\@numberedoptimization}[5][free]{
  \ifthenelse{\equal{#1}{free}}{
    \ifthenelse{\equal{#2}{}}{
      \begin{alignat}{2}
        #3_{#4}\text{ } & #5
      \end{alignat}
    }{
      \begin{alignat}{4}
        #2 & {}= & #3_{#4}\text{ } & #5
      \end{alignat}
    }
  }{
    \ifthenelse{\equal{#2}{}}{
      \begin{alignat}{2}
        #3_{#4}\text{ } & #5 \\
        \mathrm{s.t.}\text{ } & #1
      \end{alignat}
    }{
      \begin{alignat}{4}
        #2 & {}= & #3_{#4}\text{ } & #5 \\
        &     & \mathrm{s.t.}\text{ } & #1
      \end{alignat}
    }
  }
}
\newcommand{\@nonumberoptimization}[5][free]{
  \ifthenelse{\equal{#1}{free}}{
    \ifthenelse{\equal{#2}{}}{
      \begin{alignat*}{2}
        #3_{#4}\text{ } & #5
      \end{alignat*}
    }{
      \begin{alignat*}{4}
        #2 & {}= & #3_{#4}\text{ } & #5
      \end{alignat*}
    }
  }{
    \ifthenelse{\equal{#2}{}}{
      \begin{alignat*}{2}
        #3_{#4}\text{ } & #5 \\
        \mathrm{s.t.}\text{ } & #1
      \end{alignat*}
    }{
      \begin{alignat*}{4}
        #2 & {}= & #3_{#4}\text{ } & #5 \\
        &     & \mathrm{s.t.}\text{ } & #1
      \end{alignat}
    }
  }
}
